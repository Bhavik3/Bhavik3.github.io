3   /**
4    * 
5    * Whereas Singleton design pattern introduces single globally
6    * accessible object the Multiton pattern defines many globally
7    * accessible objects. The client asks for the correct instance 
8    * from the Multiton by passing an enumeration as parameter.
9    * 
10   * In this example Nazgul is the Multiton and we can ask single
11   * Nazgul from it using NazgulName. The Nazguls are statically
12   * initialized and stored in concurrent hash map.
13   *
14   */
15  public class App {
16  	
17      public static void main( String[] args ) {
18      	System.out.println("KHAMUL=" + Nazgul.getInstance(NazgulName.KHAMUL));
19      	System.out.println("MURAZOR=" + Nazgul.getInstance(NazgulName.MURAZOR));
20      	System.out.println("DWAR=" + Nazgul.getInstance(NazgulName.DWAR));
21      	System.out.println("JI_INDUR=" + Nazgul.getInstance(NazgulName.JI_INDUR));
22      	System.out.println("AKHORAHIL=" + Nazgul.getInstance(NazgulName.AKHORAHIL));
23      	System.out.println("HOARMURATH=" + Nazgul.getInstance(NazgulName.HOARMURATH));
24      	System.out.println("ADUNAPHEL=" + Nazgul.getInstance(NazgulName.ADUNAPHEL));
25      	System.out.println("REN=" + Nazgul.getInstance(NazgulName.REN));
26      	System.out.println("UVATHA=" + Nazgul.getInstance(NazgulName.UVATHA));    	
27      }
28  }
