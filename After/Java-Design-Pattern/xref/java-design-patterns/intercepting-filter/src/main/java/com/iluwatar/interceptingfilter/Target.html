17  /**
18   * This is where the requests are displayed after being validated by filters.
19   * 
20   * @author mjoshzambales
21   *
22   */
23  public class Target extends JFrame {
24  
25  	private static final long serialVersionUID = 1L;
26  
27  	private JTable jt;
28  	private JScrollPane jsp;
29  	private DefaultTableModel dtm;
30  	private JButton del;
31  
32  	public Target() {
33  		super("Order System");
34  		setDefaultCloseOperation(EXIT_ON_CLOSE);
35  		setSize(640, 480);
36  		dtm = new DefaultTableModel(new Object[] { "Name", "Contact Number",
37  				"Address", "Deposit Number", "Order" }, 0);
38  		jt = new JTable(dtm);
39  		del = new JButton("Delete");
40  		setup();
41  	}
42  
43  	private void setup() {
44  		setLayout(new BorderLayout());
45  		JPanel bot = new JPanel();
46  		add(jt.getTableHeader(), BorderLayout.NORTH);
47  		bot.setLayout(new BorderLayout());
48  		bot.add(del, BorderLayout.EAST);
49  		add(bot, BorderLayout.SOUTH);
50  		jsp = new JScrollPane(jt);
51  		jsp.setPreferredSize(new Dimension(500, 250));
52  		add(jsp, BorderLayout.CENTER);
53  
54  		del.addActionListener(new DListener());
55  
56  		JRootPane rootPane = SwingUtilities.getRootPane(del);
57  		rootPane.setDefaultButton(del);
58  		setVisible(true);
59  	}
60  
61  	public void execute(String[] request) {
62  		dtm.addRow(new Object[] { request[0], request[1], request[2],
63  				request[3], request[4] });
64  	}
65  
66  	class DListener implements ActionListener {
67  		@Override
68  		public void actionPerformed(ActionEvent e) {
69  			int temp = jt.getSelectedRow();
70  			if (temp == -1)
71  				return;
72  			int temp2 = jt.getSelectedRowCount();
73  			for (int i = 0; i < temp2; i++) {
74  				dtm.removeRow(temp);
75  			}
76  		}
77  	}
78  }
